public interface IVoar
{
    void Voar();
}

public interface INadar
{
    void Nadar();
}

public class Pato : IVoar, INadar
{
    public void Voar() { Console.WriteLine("O pato esta voando"); }
    public void Nadar() { Console.WriteLine("O pato esta nadando"); }
}

public class Aguia : IVoar
{
    public void Voar() { Console.WriteLine("A aguia esta voando"); }
}

public class Peixe : INadar
{
    public void Nadar() { Console.WriteLine("O peixe esta nadando"); }
}

public class Program
{
    public static void Main()
    {
        Pato pato = new Pato();
        Aguia aguia = new Aguia();
        Peixe peixe = new Peixe();

        pato.Voar();
        pato.Nadar();
        aguia.Voar();
        peixe.Nadar();
    }
}
