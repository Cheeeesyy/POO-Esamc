abstract class Funcionario
{
    public string Nome;
    public abstract float CalcularSalario();
}

class Gerente : Funcionario
{
    public override float CalcularSalario()
    {
        return 10000;
    }
}

class Programador : Funcionario
{
    public override float CalcularSalario()
    {
        return 5000;
    }
}

class Program
{
    static void Main()
    {
        Funcionario g = new Gerente { Nome = "Carlos" };
        Funcionario p = new Programador { Nome = "Ana" };

        Console.WriteLine(g.Nome + " recebe R$ " + g.CalcularSalario());
        Console.WriteLine(p.Nome + " recebe R$ " + p.CalcularSalario());
    }
}
